    def _load_drinks(self):
        db = shelve.open(self.db_file)
        if not db.has_key('drinks'):
            db.close()
            self.db_reset()

        else:
            self.replace_drinks(db['drinks'])
            self.replace_mixes(db['mixes'])
            db.close()

        self.monitor.start()

        if self.mix_dirty:
            self._mixes_updated()
